Docker

Основные команды Docker

docker --version Проверка версии Docker
docker ps Список запущенных контейнеров
docker ps -a Все контейнеры (включая остановленные)
docker images Список образов
docker run -it ubuntu bash Запуск контейнера Ubuntu в интерактивном режиме
docker stop container_name Остановка контейнера
docker rm container_name Удаление контейнера
docker rmi image_name Удаление образа
docker exec -it container_name bash Вход в запущенный контейнер
docker logs container_name Просмотр логов контейнера

Dockerfile пример

FROM ubuntu:20.04
RUN apt-get update && apt-get install -y nginx
COPY index.html /var/www/html/
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]

Сборка образа: docker build -t my-nginx .

Docker Compose

Основные команды

docker-compose up Запуск сервисов
docker-compose up -d Запуск в фоновом режиме
docker-compose down Остановка сервисов
docker-compose ps Статус сервисов
docker-compose logs Просмотр логов

Пример docker-compose.yml

version: '3.8'
services:
  web:
    image: nginx:alpine
    ports:
      - "80:80"
    volumes:
      - ./html:/usr/share/nginx/html
  db:
    image: postgres:13
    environment:
      POSTGRES_DB: myapp
      POSTGRES_USER: user
      POSTGRES_PASSWORD: password
    volumes:
      - db_data:/var/lib/postgresql/data

volumes:
  db_data:

Полезные команды Docker Compose

docker-compose build Пересборка образов
docker-compose exec service_name command Выполнение команды в сервисе
docker-compose restart Перезапуск сервисов
docker-compose pull Обновление образов
